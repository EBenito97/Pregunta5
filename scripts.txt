//Creamos las tablas
CREATE TABLE department(
Id int not null primary key,
name varchar(255) not null
);

CREATE TABLE employee(
id int not null primary key auto_increment,
name varchar(255),
salary int,
departmentId int,
foreign key (departmentId) references department(id)
);
